    "def preprocess_inputs(df):\n",
    "    df = df.copy()\n",
    "     # DROP TIME COLUMN\n",
    "    df= df.drop('Time', axis=1) \n",
    "    # DROP column more than 25% missng value\n",
    "    miss_value=df.columns[df.isna().mean() >=0.25]\n",
    "    df=df.drop(miss_value, axis=1)\n",
    "    # fill remaining missing value\n",
    "    for column in df.columns:\n",
    "        df[column]=df[column].fillna(df[column].mean())\n",
    "    # give text label to training example\n",
    "    df['Pass/Fail']=df['Pass/Fail'].replace({-1:\"PASS\",1:\"FAIL\"})\n",
    "    # Spilt df into X and Y\n",
    "    y=df['Pass/Fail']\n",
    "    X=df.drop('Pass/Fail',axis=1)\n",
    "    X_Train,X_Test,y_Train,y_Test=train_test_split(X,y,train_size=0.7,random_state=1)\n",
    "    # Scale X\n",
    "    scaler=StandardScaler()\n",
    "    scaler.fit(X_Train)\n",
    "    X_Train=pd.DataFrame(scaler.transform( X_Train),index=X_Train.index,columns=X_Train.columns)\n",
    "    X_Test=pd.DataFrame(scaler.transform( X_Test),index=X_Test.index,columns=X_Test.columns) \n",
    "    #return X,y\n",
    "    return  X_Train,X_Test,y_Train,y_Test\n",
    "\n"
